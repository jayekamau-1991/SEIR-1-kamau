Linux Basics

cd ~                             # Go to home directory
ls -la                           # List files with details
df -h /tmp                       # Check disk space
du -sh /tmp/*                    # Check what's using /tmp

mount | grep /tmp                # Check /tmp mount
free -h                          # Check memory
top                              # See running processes
hostnamectl                      # Show server info

Jenkins & Docker

sudo yum install jenkins -y      # Install Jenkins
sudo yum install docker -y       # Install Docker
sudo systemctl start jenkins     # Start Jenkins
sudo systemctl start docker      # Start Docker
sudo systemctl stop jenkins      # Stop Jenkins
sudo systemctl status jenkins    # Check Jenkins status
sudo systemctl enable jenkins    # Auto start on reboot
sudo systemctl restart jenkins   # Restart Jenkins
sudo usermod -aG docker jenkins  # Connect Jenkins to Docker
sudo docker run hello-world      # Test Docker works
sudo docker ps                   # List running containers
curl -I http://localhost:8080    # Test Jenkins is responding

AWS CLI

aws sts get-caller-identity      # Verify IAM Role attached
